95 inline void TopologicalMesh::interpolateNormalOnFaces( FaceHandle fh,
96  OpenMesh::FPropHandleT<Normal> fProp ) {
97  auto nph = halfedge_normals_pph();
98 
99  // init sum to first
100  auto heh = halfedge_handle( fh );
101  property( fProp, fh ) = property( nph, heh );
102  heh = next_halfedge_handle( heh );
103 
104  // sum others

106  {
107  property( fProp, fh ) += property( nph, heh );

172 template <typename T>
173 void TopologicalMesh::interpolatePropsOnFaces(
174  FaceHandle fh,
175  const std::vector<OpenMesh::HPropHandleT<T>>& hProps,
176  const std::vector<OpenMesh::FPropHandleT<T>>& fProps ) {
177  auto heh = halfedge_handle( fh );
178  const size_t v = valence( fh );
179 
180  // init sum to first
181  for ( size_t j = 0; j < fProps.size(); ++j )
182  {
183  auto hp = hProps[j];
184  auto fp = fProps[j];
185  property( fp, fh ) = property( hp, heh );
186  }
187  heh = next_halfedge_handle( heh );
188  // sum others
189  for ( size_t i = 1; i < v; ++i )
190  {
191  for ( size_t j = 0; j < fProps.size(); ++j )
192  {
193  auto hp = hProps[j];
194  auto fp = fProps[j];
195  property( fp, fh ) += property( hp, heh );
196  }
197  heh = next_halfedge_handle( heh );
198  }
199  // normalize
200  for ( size_t j = 0; j < fProps.size(); ++j )
201  {
202  auto fp = fProps[j];
203  property( fp, fh ) = property( fp, fh ) / v;
204  }
205 }
